使用Java新手入门打星星（三角形)
private class{
  public static void main(String[] args){
    for(int i=1;i<5;i++){
      for(int j=0;j<4-i;j++){
      System.out.print(" ");
      }
      for(int l=0;l<i+i-1;l++){
      System.out.print("*");
      }
      Systemout.println();
    }
  }
}
